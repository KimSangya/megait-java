**Student 클래스 만들기** 
필드 : 이름, 국, 영, 수, 평균, 등급(gpa), 학년(grade) --> private
메서드 : 
	1) 생성자 
		Student()
		Student(이름)
		Student(이름, 국, 영, 수)
	2) getter
	3) setter
		setName(String name) ~> 10글자 이하만 저장
		setKr(int kr), setEn(), setMa() ~> 0점 이상 100점 이하만 저장
		setAvg() ~> 국, 영, 수 필드로 평균 계산 및 avg 필드에 저장
		setGqa() ~> 평균 가지고 A, B, C, D, F 중 1개를 gpa 필드에 저장
		setGrade() ~> 1 ~ 4 학년만 저장
	4) getInfo() 모든 정보를 1개의 String 형태로 return 메인메서드는 마음대로..



#### student

```java
public class Student {
	private String name;
	private int kr;
	private int en;
	private int ma;
	private double avg;
	private String gpa;
	private String info;
	private int grade;

	Student() {
	}

	Student(String name) {
	}

	Student(String name, int kr, int en, int ma, int ga) {
		setName(name);
		setKr(kr);
		setEn(en);
		setMa(ma);
		setAvg();
		setGpa();
		setGrade(ga);
	}

	public String getName() {
		return name;
	}

	// setName(String name) ~> 10글자 이하만 저장
	public void setName(String name) {
		if (name.length() < 10) {
			this.name = name;
		}
	}

	public int getKr() {
		return kr;
	}

	// setKr(int kr), setEn(), setMa() ~> 0점 이상 100점 이하만 저장
	public void setKr(int kr) {
		if (kr <= 100 && kr >= 0) {
			this.kr = kr;
		}
	}

	public int getEn() {
		return en;
	}

	public void setEn(int en) {
		if (en <= 100 && en >= 0) {
			this.en = en;
		}
	}

	public int getMa() {
		return ma;
	}

	public void setMa(int ma) {
		if (ma <= 100 && ma >= 0) {
			this.ma = ma;
		}
	}

	public double getAvg() {
		return avg;
	}

	public void setAvg() {
		double avg;
		avg = (kr + en + ma) / 3.0;
		this.avg = avg;
	}

	public String getGpa() {
		return gpa;
	}

	public void setGpa() {
		switch ((int) this.avg / 10) {
		case 10:
		case 9:
			this.gpa = "A";
			break;
		case 8:
			this.gpa = "B";
			break;
		case 7:
			this.gpa = "C";
			break;
		case 6:
			this.gpa = "D";
			break;
		default:
			this.gpa = "F";
		}
	}

	public int getGrade() {
		return grade;
	}

	public void setGrade(int ga) {
		if (ga >= 1 && ga <= 4) {
			this.grade = ga;
		}
	}

	public String getInfo() {
		return info;
	}

	public String setInfo() {
		String info;
		info = "이름 : " + this.name + " 학년 : " + this.grade + " 국어 : " + this.kr + " 국어 : " + this.en + " 국어 : "
				+ this.ma + " 평균 : " + this.avg + " 성적 : " + this.gpa;
		this.info = info;
		return info;
	}

}
```
