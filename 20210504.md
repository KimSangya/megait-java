### 5.4일 과제

Pokemon 4마리 객체 생성
	1. name 저장
	

 2. "피카츄", "라이츄", "파이리", "꼬부기", "푸린", "치코리타"
    중 1개를 랜덤하게 선택해서 저장 (중복 이름 가능)	 

3. level 저장
   1 ~ 100 중 랜덤하게 저장

4. hp 저장
   70% 확률로 level의 100배, 30% 확률로 level의 150배 저장

5. 저장된 4개의 포켓몬 인스턴스의 모든 정보 출력

   ##### 메인 메서드

```java
package com.day11.homework;

public class Homework01 {
	public static void main(String[] args) {
		Pokemon p1,p2,p3,p4;		
		p1 = new Pokemon();
		p2 = new Pokemon();
		p3 = new Pokemon();
		p4 = new Pokemon();
		int a,b,c,d,e,f,g,h,i,j,k,l;
		
		String[] pokename; 
		pokename = new String[] { "피카츄", "라이츄", "파이리", "꼬부기", "푸린", "치코리타"};
		 // 1번
			 	a = (int)(Math.random()*5);
			 		p1.name = pokename[a];
			 	b = (int)(Math.random()*5);
			 		p2.name = pokename[b];
			 	c = (int)(Math.random()*5);
			 		p3.name = pokename[c];
			 	d = (int)(Math.random()*5);
			 		p4.name = pokename[d];
		// 2번	 		
			 	e = (int)(Math.random()*100)+1;
			 		p1.level = e;
			 	f = (int)(Math.random()*100)+1;
			 		p2.level = f;
			 	g = (int)(Math.random()*100)+1;
			 		p3.level = g;
			 	h = (int)(Math.random()*100)+1;
			 		p4.level = h;
		// 3번
			 i = (int)(Math.random()< 0.3 ? p1.level*150 : p1.level*100);
			 		p1.hp = i;
			 j = (int)(Math.random()< 0.3 ? p2.level*150 : p2.level*100);
			 		p2.hp = j;
			 k = (int)(Math.random()< 0.3 ? p3.level*150 : p3.level*100);
			 		p3.hp = k;
			 l = (int)(Math.random()< 0.3 ? p4.level*150 : p4.level*100);
			 		p4.hp = l;
			 		
		//4번	 		
		System.out.println("1번 포켓몬 : " + p1.name + ", 레벨:" + p1.level + ", hp:"+ p1.hp);
		System.out.println("2번 포켓몬 : " + p2.name + ", 레벨:" + p2.level + ", hp:"+ p2.hp);
		System.out.println("3번 포켓몬 : " + p3.name + ", 레벨:" + p3.level + ", hp:"+ p3.hp);
		System.out.println("4번 포켓몬 : " + p4.name + ", 레벨:" + p4.level + ", hp:"+ p4.hp);
		
	}
}
```
