import java.util.ArrayList;
import java.util.List;

import javax.swing.JOptionPane;
class Thread1 extends Thread {
	boolean run = true; 
	
	public void terminate() {
		run = false;
	}
	
	@Override
	public void run(){
		int i = 1;
		while(run) {
			System.out.println("룰루 랄라");
			try {
				Thread.sleep(3000);
			} catch (InterruptedException e) {
				e.printStackTrace();
			}
			if(i < 10) {
				++i;
			}
			else {
				terminate();
			}
		}
	}
}

	// 메인스레드:구구단 랜덤하게 5문제를 냅니다.
	public static void main(String[] args) {
		long start, end;
		Thread1 th = new Thread1();
		Thread2 ti = new Thread2();
		th.start();
		int answer = 0;
		start = System.currentTimeMillis();
				
		for(int i = 0; i < 5 ; ++i) {
			int	a = (int)((Math.random()*8)+2);
			int	b = (int)((Math.random()*9)+1);
			String qustion = a + " X " + b + " = " + "?";
			String d = JOptionPane.showInputDialog(qustion);
			int c = Integer.parseInt(d);
			
		if( c == a*b) {
			end = System.currentTimeMillis();
			JOptionPane.showMessageDialog(null, "정답!!");
			++answer;
			
			System.out.println("맞춘 시간 : " + ((end - start)/1000) + "초");
			ti.records.add((int) ((end - start)/1000));
			}
		
		else {
			JOptionPane.showMessageDialog(null, "땡..");
			}
		
		}
		for (int i = 0; i < ti.records.size(); ++i) {
			System.out.println(i + " 번 문제" + ti.records );
