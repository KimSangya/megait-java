package com.day27.quiz;

import java.awt.Dimension;
import java.awt.GridLayout;
import java.awt.Point;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JOptionPane;

public class Quiz01 {
	int i = 1;
	private JFrame frame;
	private JButton btn[] = new JButton[16];

	Quiz01() {
		frame = new JFrame("1~16까지 누르기 시작.");

		frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		frame.setSize(new Dimension(800, 800));
		frame.setLocationRelativeTo(null);

		for (int i = 0; i < btn.length; ++i) {
			btn[i] = new JButton(String.valueOf(i + 1));
		}
		// 원소들을 재배치를 해야한다.
		// 1번
		for (int i = 0; i < 20; ++i) {
			int i1 = (int) (Math.random() * btn.length);
			int i2 = (int) (Math.random() * btn.length);

			JButton tmp = btn[i1];
			btn[i1] = btn[i2];
			btn[i2] = btn[3];
			btn[3] = btn[4];
			btn[4] = btn[5];
			btn[5] = btn[6];
			btn[6] = btn[7];
			btn[7] = btn[8];
			btn[8] = btn[9];
			btn[9] = btn[10];
			btn[10] = btn[11];
			btn[11] = btn[12];
			btn[12] = btn[13];
			btn[13] = btn[14];
			btn[14] = btn[15];
			btn[15] = tmp;

		}
		for (int i = 0; i < btn.length; ++i) {
			frame.add(btn[i]);
		}
		// 2번
		/*
		 * List<Integer> list = new ArrayList<Integer>(16); //리스트에 1~16을 순차적으로 넣기.
		 * for(int i = 0; i <btn.length; ++i) { list.add(i+1); }
		 * 
		 * //list에 원소를 제비뽑기 해서 button들에게 넣어주기. for(int i = 0; i < btn.length; ++i) { int
		 * n = list.get((int)(Math.random() * list.size())); list.remove((Integer)n);
		 * btn[i] = new JButton(String.valueOf(n)); }
		 */
		ActionListener listener = new ActionListener() {

			@Override
			public void actionPerformed(ActionEvent e) {
				JButton clickedBtn = (JButton)e.getSource();
				
				if(Integer.parseInt(clickedBtn.getText())== (i)) {
					JOptionPane.showMessageDialog(frame, clickedBtn.getText()+ "번을 클릭하셨습니다!");
					++i;
					clickedBtn.setEnabled(false);
				}
				else {
					JOptionPane.showMessageDialog(frame, "땡 틀렸습니다 LOSE!");
					frame.dispose();
					new Quiz01();
				}
				if(i == 17) {
					JOptionPane.showMessageDialog(frame, "축하합니다. 당신이 이겼습니다 WIN!");
					frame.dispose();
				}
				
				
			}
		};
		for (int i = 0; i < btn.length; ++i) {
			btn[i].addActionListener(listener);
			}
		frame.setLayout(new GridLayout(4, 4));
		frame.setVisible(true);
	}

	public static void main(String[] args) {
		new Quiz01();
	}
}
