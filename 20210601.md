package com.day29.homework;

import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileWriter;
import java.io.IOException;
import java.util.Scanner;

/*
   구구단 2단.txt ~ 9단.txt 을 모두 저장해보자.
	2단.txt
	2 X 1 = 2
	2 X 2 = 4
	..
	2 X 9 = 18
---------------------------------------
	위에서 만든 파일 사용하여 
	사용자에게 '단'을 입력 받고 해당 파일을 출력
	이중포문 사용을 하는게 나을듯..?
 */
public class Homework01 {
	public static void main(String[] args) {
		int num;
		try {
			
			for(int i = 2; i <= 9; ++i) {
				FileWriter Fw = new FileWriter("C:\\Users\\김건호\\Desktop\\"+i+".txt");
				for(int j = 1; j <= 9; ++j) {
					Fw.write(i + " X " + j + " = " + i*j + "\n");
				}
				Fw.close();
			}
			
			
		} catch (IOException e) {
			e.printStackTrace();
		}
		Scanner sc1 = new Scanner(System.in);
		System.out.println("찾고 싶은 구구단의 숫자를 입력하세요.");
		num = sc1.nextInt();
		try {
			FileInputStream fIn = new FileInputStream("C:\\Users\\김건호\\Desktop\\"+num+".txt");
			Scanner sc2 = new Scanner(fIn);
			String s;
			while(sc2.hasNext()) {
				s = sc2.nextLine();
				System.out.println(s);
			}
			sc2.close();
			sc1.close();
			fIn.close();
		} catch (IOException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
	}
}
