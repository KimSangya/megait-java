package day01.quiz;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;
import java.util.Scanner;

/*
 * Quiz03.
 * 회원의 username을 입력받고
 * 해당 username의 적립금을 5000원 추가하기.
 */
public class Quiz03 {
	public static void main(String[] args) {
		String name; 
		String username = "root";
		String password = "1234";
		String driverClassName = "org.mariadb.jdbc.Driver";
		Scanner sc = new Scanner(System.in);
		System.out.println("회원의 닉네임을 입력하세요");
		name = sc.next();
		String dbUrl = "jdbc:mysql://127.0.0.1:3306/studyDB?useUnicode=true&characterEncoding=utf8";
		// 드라이버 클래스 로드
		String sql = "UPDATE member SET point = point + 5000 WHERE username = '"+name+"'";
		Connection conn = null;
		PreparedStatement ps = null;
		try {
			Class.forName(driverClassName);
		} catch (ClassNotFoundException e) {
			e.printStackTrace();
		}
		try {
			conn = DriverManager.getConnection(dbUrl,username,password);
			ps = conn.prepareStatement(sql);
			ps.execute();
		} catch (SQLException e) {
			e.printStackTrace();
		} finally{
			// conn, ps 닫기
			
			
			try {
				if(ps != null) 
				ps.close();
				if(conn != null)
				conn.close();
			} catch (SQLException e) {
				// TODO Auto-generated catch block
				e.printStackTrace();
			}
		
		
		
		// 명령문 준비 및 실행
		
		System.out.println("실행완료!");
		}
		
		
	}
}
